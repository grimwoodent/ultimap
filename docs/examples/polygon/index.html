import { geo, Strategy } from 'ultimap';

const osmGeo = geo.byStrategy(new Strategy.Leaflet());
osmGeo.map.create(holder_element, {
    center: [57.767131, 40.928349],
    zoom: 14,
}).load().then((map) => {
    osmGeo.polygon.create([[
        [57.769131, 40.93534],
        [57.765131, 40.93534],
        [57.765131, 40.91834],
        [57.769131, 40.91834],
    ]], {}).addTo(map);
});

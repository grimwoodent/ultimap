import { geo, Strategy } from 'ultimap';

geo.map.byStrategy(new Strategy.Leaflet()).create(holder_element, {
    center: [57.767131, 40.928349],
    zoom: 16,
}).load().then((map) => {
    geo.marker.create(map.getCenter(), {
        icon: {
            src: './image/map-marker-black.png',
            size: [32, 32],
            offset: [16, 32],
        },
    }).addTo(map);
});
